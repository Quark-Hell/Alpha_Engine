macro(recurse_add_cpp_items src return_list)
    file(GLOB_RECURSE new_files "${src}/*.cpp" "${src}/*.c")
    
    set(IGNORE_PATTERNS
        "/[Ee]xternal/"
        "/[Tt]hird[Pp]arty/"
        "/[Vv]endor/"
        "/.git/"
    )
    
    set(filtered_files)
    foreach(file ${new_files})
        string(REPLACE "\\" "/" normalized_file "${file}")
        
        set(should_ignore FALSE)
        foreach(pattern ${IGNORE_PATTERNS})
            if(normalized_file MATCHES "${pattern}")
                set(should_ignore TRUE)
                break()
            endif()
        endforeach()
        
        if(NOT should_ignore)
            list(APPEND filtered_files ${file})
        endif()
    endforeach()
    
    list(APPEND ${return_list} ${filtered_files})
    
    list(LENGTH filtered_files count)
endmacro()

function(add_cpp_items src return_list)
file(GLOB templateFiles RELATIVE ${src} "${src}/*.cpp")
    foreach(file ${templateFiles})
		set(copy_src "${src}/${file}")
		list(APPEND return_list "${copy_src}")
    endforeach()
	message("${return_list}\n")
	set(${return_list} "${${return_list}}" PARENT_SCOPE)
endfunction()

function(add_definition ModuleName)
	set(ExportModule
			"add_compile_definitions(${ModuleName})\n"
	)

	file(APPEND "${Build_Dir}/config.cfg"
			${ExportModule}
	)
endfunction()

function(copy_file src dest)
	message("${Cyan}copy file${ColourReset} ${src} to\n ${dest}\n")

	execute_process(
	COMMAND ${CMAKE_COMMAND} -E copy_if_different
	${src}
	${dest}
	)
endfunction()

function(recurse_copy_headers src dest)
    message("${Red}copy headers${ColourReset}")

    cmake_parse_arguments(ARG "" "" "IGNORE_PATTERNS" ${ARGN})

    if(NOT DEFINED ARG_IGNORE_PATTERNS)
        set(ARG_IGNORE_PATTERNS
            "/.git/"
        )
    endif()
    
    file(GLOB_RECURSE templateFiles "${src}/*.h" "${src}/*.hpp")
    
    set(filtered_files)
    foreach(file ${templateFiles})
        string(REPLACE "\\" "/" normalized_file "${file}")
        
        set(should_ignore FALSE)
        foreach(pattern ${ARG_IGNORE_PATTERNS})
            if(normalized_file MATCHES "${pattern}")
                set(should_ignore TRUE)
                break()
            endif()
        endforeach()
        
        if(NOT should_ignore)
            list(APPEND filtered_files ${file})
        endif()
    endforeach()
    
    set(copied_count 0)
    set(skipped_count 0)
    
    foreach(file ${filtered_files})
        file(RELATIVE_PATH rel_path ${src} ${file})
        set(copy_dest "${dest}/${rel_path}")
        
        get_filename_component(dest_dir ${copy_dest} DIRECTORY)
        file(MAKE_DIRECTORY ${dest_dir})
        
        message("${Cyan}copy header${ColourReset} ${file} to\n  ${copy_dest}")
        
        execute_process(
            COMMAND ${CMAKE_COMMAND} -E copy_if_different ${file} ${copy_dest}
        )
    endforeach()
    
    # Statistics
    list(LENGTH templateFiles total_found)
    list(LENGTH filtered_files total_processed)
    math(EXPR ignored_count "${total_found} - ${total_processed}")
    
    if(ignored_count GREATER 0)
        message(STATUS "Copied ${copied_count} headers (${ignored_count} ignored from External)")
    else()
        message(STATUS "Copied ${copied_count} headers")
    endif()
    
    message("\n")
endfunction()

function(recurse_copy_assets src dest)
	message("\n${Red}copy assets${ColourReset}")
	file(GLOB_RECURSE templateFiles RELATIVE ${src} "${src}/*")
	foreach(templateFile ${templateFiles})
		set(copy_src "${src}/${templateFile}")
		set(copy_dest "${dest}/${templateFile}")

		message("${Cyan}copy asset${ColourReset} ${copy_src} to\n ${copy_dest}")

		execute_process(
		COMMAND ${CMAKE_COMMAND} -E copy_if_different
		${copy_src}
		${copy_dest}
		)
	endforeach(templateFile)
	message("\n")
endfunction()
